# input function
simulate_for_different_beta <- function(beta_values, network, static = TRUE) {
    fraction_recovered <- numeric(length = length(beta_values))
    for (i in seq_along(beta_values)) {
        if (static) {
            df <- simulate_seir_static(network, beta=beta_values[i])}
        else {
            df <- simulate_seir_temporal(network, beta=beta_values[i])
        }
        R_final <- df[nrow(df), "R"]
        N <- sum(df[nrow(df),c("S", "E", "I", "R") ])

        fraction_recovered[i] <- R_final /N
    }

    return(cbind(beta_values,fraction_recovered))
}
